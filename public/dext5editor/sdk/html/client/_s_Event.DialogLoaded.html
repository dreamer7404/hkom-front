DEXT5 Editor :: Event.DialogLoaded

config.Event.DialogLoaded

callback 이벤트를 설정합니다.

remarks

dext_editor_dialog_loaded_event를 설정합니다.
각종 dialog가 open되었을시 이벤트가 발생합니다.
config.Event에 Object로 Event들을 모아서 설정이 가능합니다.

sample code

<script type="text/javascript" src="dext5editor/js/dext5editor.js"></script>	
 
<!-- ..... 생략 ..... -->
 
<div style="width:900px;height:550px">    
    <script type="text/javascript">
        // 이벤트발생시 호출할 함수를 설정합니다.
        var eventFunc = function(eventParam1, eventParam2, eventParam3){
            alert("Event!");
            // work something ...
        }

        // 각종 dialog가 open되었을시 이벤트가 발생합니다. (Object로 모아서 설정이 가능합니다.)
        DEXT5.config.Event = {
            DialogLoaded : eventFunc,
            // Add Another Event Config ...
        }

        /* 다음과 같이 설정도 가능합니다.
        DEXT5.config.Event.DialogLoaded = eventFunc;
        */

        new Dext5editor("editor1");
    </script>       
</div>
